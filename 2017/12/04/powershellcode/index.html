1.查询值

Get-ADUser -Filter {pager -eq "***"}

2.查询导出

Get-ADUser -Filter * -Properties * -SearchBase "OU=GZ,DC=CONTOSO,DC=COM" | Select-Object name,department |Export-Csv -NoTypeInformation -Encoding UTF8 -Path "d:\dep.csv"

3.修改值

Set-ADUser *** -DisplayName ****
get-ADUser -Filter {employeeid -eq "00000"}|Set-ADUser -employeeid 00001
Set-ADUser *** -replace @{pager ="ABC"}
Import-Csv -Path c:\abc.csv | foreach {Set-ADUser -Identity $_.name -DisplayName $_.DisplayName}
Import-Csv -Path d:\dep.csv | foreach {Set-ADUser $_.name -replace @{telephonenumber = $_.telephonenumber}}

4.批量导入查询导出

Import-Csv -Path d:\1.csv | foreach {Get-ADUser $_.name -Properties * } |select name,LogonWorkstations|Export-Csv -NoTypeInformation -Encoding UTF8 -Path "d:\dep.csv"

Import-csv -Path d:\1.csv | ForEach-Object {
    $ID = $_.employeeid
    get-aduser -filter {employeeid -eq $ID} -Properties employeeid,cn,description,mail | select employeeid,cn,description,mail  >>d:4.txt
}

5.新建用户

New-ADUser -Name ken -DisplayName ken -SamAccountName ken -Company canway -UserPrincipalName ken -AccountPassword (ConvertTo-SecureString 1qaz@WSX -AsPlainText -Force) -Enabled $true -Path "ou=it,ou=gz,dc=contoso,dc=com"

6.批量新建用户

Import-csv d:\0.csv | foreach {New-ADUser -Name $_.name -description $_.FullChineseName -givenName $_.FirstName -DisplayName $_.name -Samaccountname $_.name -UserPrincipalName $_.un -Surname $_.LastName -Company $_.StaffCompany -officephone $_.officephone -employeeid $_.StaffNo -department $_.Department  -manager $_.SuperiorManager -title $_.AccountTitle -AccountPassword (ConvertTo-SecureString Kaifa@123 -AsPlainText -Force) -Path "OU=MDBU1_Users,OU=MDBU1,OU=kfft,DC=gwkf,DC=cn" -ChangePasswordAtlogon $True –Enabled $True}

Import-csv d:\0.csv | foreach {New-ADUser -Name $_.name -description $_.description -givenName $_.givenName -DisplayName $_.name -Samaccountname $_.name -UserPrincipalName $_.un -Surname $_.sn -Company $_.company  -title $_.title -AccountPassword (ConvertTo-SecureString Kaifa@123 -AsPlainText -Force) -Path "OU=CCBU_Users,OU=CCBU,OU=KFPI,DC=gwkf,DC=cn" -ChangePasswordAtlogon $True –Enabled $True}

Import-csv c:\ADuser.csv | foreach {New-ADUser -Name $_.Name -DisplayName $_.Displayname -Samaccountname $_.Samaccountname -UserPrincipalName $_.UserPrincipalName -Company $_.Company -Department $_.Department -Surname $_.Surname -GivenName $_.GivenName -AccountPassword (ConvertTo-SecureString $_.AccountPassword -AsPlainText -Force) -Path $_.Path -ChangePasswordAtlogon $True –Enabled $True}

Import-csv d:\A.csv|foreach{get-adcomputer $_.computer -properties * }|Select-Object name,operatingsystem |Export-Csv -NoTypeInformation -Encoding UTF8 -Path "d:\c.csv"

7.启用Exchange

Enable-Mailbox -Identity *** -Alias '***' -Database '*****'

8.查询Exchange用户信息

Get-MailboxStatistics '*****' | fl

9.迁移Exchange用户数据库

"***" | New-MoveRequest -TargetDatabase '*****'

10.启用Skype

Enable-CsUser –Identity ShuangShuangGuan -RegistrarPool ****.***.cn -SipAddressType UserPrincipalName

11.批量删除计算机

Import-Csv -Path d:\1.csv | foreach {Get-ADComputer $_.ComputerName  | Remove-ADObject -Recursive -Confirm:$False }
